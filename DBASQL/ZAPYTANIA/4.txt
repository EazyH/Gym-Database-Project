SELECT KLUB.ID_KLUBU, SALA.NR_SALI, TYP_ZAJECIA.NAZWA_TYPUZ, ZAJECIE.CZAS_ODBYCIA, PRACOWNIK.IMIE, PRACOWNIK.NAZWISKO
FROM KLUB LEFT JOIN SALA ON KLUB.ID_KLUBU=SALA.ID_KLUBU
RIGHT JOIN ZAJECIE ON SALA.ID_SALI=ZAJECIE.ID_SALI
LEFT JOIN TYP_ZAJECIA ON ZAJECIE.ID_TYPUZ=TYP_ZAJECIA.ID_TYPUZ
LEFT JOIN PRACOWNIK ON ZAJECIE.PESEL=PRACOWNIK.PESEL
WHERE ZAJECIE.CZAS_ODBYCIA > '2016-10-01'
ORDER BY ZAJECIE.CZAS_ODBYCIA ASC
