SELECT KLUB.ID_KLUBU, COUNT(*) AS LICZBA_TRENEROW_FITNESS
FROM KLUB LEFT JOIN PRACOWNIK ON KLUB.ID_KLUBU=PRACOWNIK.ID_KLUBU
LEFT JOIN STANOWISKO ON PRACOWNIK.ID_STANOWISKA=STANOWISKO.ID_STANOWISKA
WHERE STANOWISKO.NAZWA_STANOWISKA='Instruktor fitness' 
GROUP BY KLUB.ID_KLUBU
ORDER BY LICZBA_TRENEROW_FITNESS DESC
